FUNCTION SortTasks
    DISPLAY "Sort by: "
    DISPLAY "1. Due Date"
    DISPLAY "2. Priority"
    INPUT sortChoice
    SWITCH sortChoice
        CASE "1":
            SORT TaskList BY dueDate
        CASE "2":
            SORT TaskList BY priority
        DEFAULT:
            DISPLAY "Invalid choice."
    END SWITCH
END FUNCTION

/**
 * Aufgabe 5)
 * Lösung 
 */

funktion SortTasks(array TaskList)
    ZEIGE "Sort by: "
    ZEIGE "1. Due Date"
    ZEIGE "2. Priority"
    string sortChoice := ""
    EINGABE sortChoice
    wenn sortChoice = "1" dann
       SORTIERE TaskList NACH dueDate
    sonst wenn sortChoice = "2"
       SORTIERE TaskList NACH priority
    sonst
        ZEIGE "Invalid choice."
    Ende wenn
Ende funktion
